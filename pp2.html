import pandas as pd
import scipy.io as sio
import os
import numpy as np

# Diretório com os arquivos de Treino
directory = "/content/drive/MyDrive/Colab Notebooks/Datasets/Fall Detection/Train/"

# Lista para armazenar as estatísticas de cada arquivo
data_rows = []

# Lista de features para calcular estatísticas
features = ["acc_x", "acc_y", "acc_z", "gyr_x", "gyr_y", "gyr_z", "mag_x", "mag_y", "mag_z"]

# Loop pelos arquivos no diretório
for filename in os.listdir(directory):
    if filename.endswith(".mat"):
        # Ler o arquivo .mat
        file_path = os.path.join(directory, filename)
        mat = sio.loadmat(file_path)
        data = mat["newData"]  # Dados na chave "newData"

        # Converter para DataFrame
        try:
            df = pd.DataFrame(data, columns=["time", "acc_x", "acc_y", "acc_z",
                                            "gyr_x", "gyr_y", "gyr_z",
                                            "mag_x", "mag_y", "mag_z", "zero"])
        except Exception as e:
            print(f"Erro ao criar DataFrame para {filename}: {e}")
            continue

        # Calcular estatísticas para cada feature
        row = {}
        for feature in features:
            row[f"{feature}_mean"] = df[feature].mean()
            row[f"{feature}_std"] = df[feature].std()
            row[f"{feature}_max"] = df[feature].max()
            row[f"{feature}_min"] = df[feature].min()

        # Adicionar o nome do arquivo
        row["filename"] = filename

        # Extrair a classe do nome do arquivo (ex.: primeiro número antes do primeiro ponto)
        if filename.split('.')[0] == 2:
            row["class_label"] = 1
        else:
            row["class_label"] = 0

        # Adicionar a linha ao conjunto de dados
        data_rows.append(row)

# Criar o DataFrame final
df_train = pd.DataFrame(data_rows)

# Definir as colunas na ordem especificada, incluindo filename e class_label
df_train.columns = [
    "acc_x_mean", "acc_x_std", "acc_x_max", "acc_x_min",
    "acc_y_mean", "acc_y_std", "acc_y_max", "acc_y_min",
    "acc_z_mean", "acc_z_std", "acc_z_max", "acc_z_min",
    "gyr_x_mean", "gyr_x_std", "gyr_x_max", "gyr_x_min",
    "gyr_y_mean", "gyr_y_std", "gyr_y_max", "gyr_y_min",
    "gyr_z_mean", "gyr_z_std", "gyr_z_max", "gyr_z_min",
    "mag_x_mean", "mag_x_std", "mag_x_max", "mag_x_min",
    "mag_y_mean", "mag_y_std", "mag_y_max", "mag_y_min",
    "mag_z_mean", "mag_z_std", "mag_z_max", "mag_z_min",
    "filename", "class_label"
]

df_train.drop(columns=["filename"], inplace=True)
